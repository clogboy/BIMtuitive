def debug_full_ifc_tree(model):
    projects = model.by_type('IfcProject')
    
    for project in projects:
        print(f"\n[PROJECT] {project.Name} | #{project.id()}")
        
        def traverse(parent, indent=0):
            tabs = "  " * (indent + 1)
            
            # A. Hiërarchische structuur (Aggregatie)
            for rel in getattr(parent, 'IsDecomposedBy', []):
                for obj in rel.RelatedObjects:
                    print(f"{tabs}[{obj.is_a().upper()}] {obj.Name or 'Geen Naam'} | #{obj.id()}")
                    
                    # B. Fysieke elementen in deze container
                    if hasattr(obj, 'ContainsElements'):
                        for rel_cont in obj.ContainsElements:
                            for el in rel_cont.RelatedElements:
                                print(f"{tabs}  >> [ELEMENT] {el.is_a()} | {el.Name or ''} | #{el.id()}")
                                
                                # C. PropertySets ophalen
                                psets = ifcopenshell.util.element.get_psets(el)
                                for pset_name, properties in psets.items():
                                    prop_list = ", ".join([f"{k}: {v}" for k, v in properties.items()])
                                    print(f"{tabs}     PSET {pset_name} -> {prop_list}")
                    
                    # Recursie
                    traverse(obj, indent + 1)

        traverse(project)
